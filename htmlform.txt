  function listarCamposRequeridosFaltantes(form) {

  window.lqEnviar = function() {
    var nombre   = document.getElementById("lq-nombre").value.trim();
    var apellido = document.getElementById("lq-apellido").value.trim();
    var ciudad   = document.getElementById("lq-ciudad").value.trim();
    var telefono = document.getElementById("lq-telefono").value.trim();
    var email    = document.getElementById("lq-email").value.trim();
    var fecha    = document.getElementById("lq-fecha").value;
    var evento   = document.getElementById("lq-evento").value;
    var mensaje  = document.getElementById("lq-mensaje").value.trim();

    if (!nombre || !apellido || !email || !telefono || !evento) {
      alert("Por favor completa los campos obligatorios: Nombre, Apellido, Teléfono, Email y Tipo de evento.");
      return;
    }

    var btn = document.getElementById("lq-send-btn");
    btn.disabled = true;
    btn.textContent = "Enviando...";

    var errEl = document.getElementById("lq-error");
    if (errEl) {
      errEl.style.display = "none";
      errEl.textContent = "";
    }

    var productoTitulo = (
      document.querySelector(".ProductItem-details-title h1") ||
      document.querySelector(".product-title h1") ||
      document.querySelector("h1.entry-title")
    );
    var producto = productoTitulo ? productoTitulo.innerText.trim() : window.location.pathname;

    var values = {
      nombre: nombre,
      apellido: apellido,
      ciudad: ciudad,
      telefono: telefono,
      email: email,
      fecha: fecha,
      evento: evento,
      mensaje: mensaje,
      producto: producto
    };

    enviarPorFormularioSquarespace(values)
    .then(function() {
      document.getElementById("lq-form-wrap").style.display = "none";
      document.getElementById("lq-success").style.display = "block";
      setTimeout(lqClose, 3500);
    })
    .catch(function(err) {
      if (errEl) {
        var msg = (err && err.message ? err.message : "").trim();
        errEl.textContent = msg ? msg : "No pudimos enviar la solicitud. Intenta nuevamente.";
        errEl.style.display = "block";
      } else {
        alert("No pudimos enviar la solicitud. Intenta nuevamente.");
      }
    })
    .finally(function() {
      btn.disabled = false;
      btn.textContent = "Enviar solicitud";
    });
  };

})();
(function () {
  var tries = 0;
  var maxTries = 30;

  function traducir() {
    var root = document.querySelector(".newsletter-form-body");
    if (!root) {
      if (tries++ < maxTries) setTimeout(traducir, 400);
      return;
    }

    var el;

    el = root.querySelector(".field.first-name .newsletter-form-field-label");
    if (el) el.textContent = "Nombre";
    el = root.querySelector(".field.last-name .newsletter-form-field-label");
    if (el) el.textContent = "Apellido";
    el = root.querySelector('label[for^="email-"].newsletter-form-field-label');
    if (el) el.textContent = "Correo electrónico";

    el = root.querySelector('input[name="fname"]');
    if (el) el.placeholder = "Nombre";
    el = root.querySelector('input[name="lname"]');
    if (el) el.placeholder = "Apellido";
    el = root.querySelector('input[name="email"]');
    if (el) el.placeholder = "Correo electrónico";

    el = root.querySelector(".newsletter-form-button-label");
    if (el) el.textContent = "Lo quiero";
    el = root.querySelector(".newsletter-form-button");
    if (el) el.value = "Lo quiero";
  }

  traducir();
})();
