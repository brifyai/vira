<script>
(function() {

  function insertarBoton() {
    var addToCartBtn = document.querySelector(".sqs-add-to-cart-button");
    if (!addToCartBtn) return;

    if (document.getElementById("lq-trigger-btn")) return;

    var btn = document.createElement("button");
    btn.id = "lq-trigger-btn";
    btn.type = "button";
    btn.textContent = "Lo quiero";
    btn.addEventListener("click", function(e) {
      e.preventDefault();
      e.stopPropagation();
      lqAbrir();
    });

    var insertTarget = addToCartBtn.closest(".sqs-add-to-cart-button-wrapper") || addToCartBtn.parentNode;
    if (!insertTarget || !insertTarget.parentNode) return;
    insertTarget.parentNode.insertBefore(btn, insertTarget);
  }

  document.addEventListener("DOMContentLoaded", insertarBoton);
  setTimeout(insertarBoton, 800);
  setTimeout(insertarBoton, 1600);

  window.lqAbrir = function() {
    document.getElementById("lq-overlay").classList.add("lq-open");
    document.body.style.overflow = "hidden";
  };

  window.lqClose = function() {
    document.getElementById("lq-overlay").classList.remove("lq-open");
    document.body.style.overflow = "";
  };

  window.lqEnviar = function() {
    var nombre   = document.getElementById("lq-nombre").value.trim();
    var apellido = document.getElementById("lq-apellido").value.trim();
    var ciudad   = document.getElementById("lq-ciudad").value.trim();
    var telefono = document.getElementById("lq-telefono").value.trim();
    var email    = document.getElementById("lq-email").value.trim();
    var fecha    = document.getElementById("lq-fecha").value;
    var evento   = document.getElementById("lq-evento").value;
    var mensaje  = document.getElementById("lq-mensaje").value.trim();

    if (!nombre || !apellido || !email || !telefono || !evento) {
      alert("Por favor completa los campos obligatorios: Nombre, Apellido, Teléfono, Email y Tipo de evento.");
      return;
    }

    var btn = document.getElementById("lq-send-btn");
    btn.disabled = true;
    btn.textContent = "Enviando...";

    var productoTitulo = (
      document.querySelector(".ProductItem-details-title h1") ||
      document.querySelector(".product-title h1") ||
      document.querySelector("h1.entry-title")
    );
    var producto = productoTitulo ? productoTitulo.innerText.trim() : window.location.pathname;

    var formData = {
      "formId":    "lq-contact-form",
      "pagePath":  window.location.pathname,
      "fields": [
        { "rawLabel": "Nombre",         "value": nombre },
        { "rawLabel": "Apellido",       "value": apellido },
        { "rawLabel": "Ciudad",         "value": ciudad },
        { "rawLabel": "Teléfono",       "value": telefono },
        { "rawLabel": "Email",          "value": email },
        { "rawLabel": "Fecha Evento",   "value": fecha },
        { "rawLabel": "Tipo de Evento", "value": evento },
        { "rawLabel": "Producto",       "value": producto },
        { "rawLabel": "Mensaje",        "value": mensaje }
      ]
    };

    fetch("/api/form-submission", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "X-Requested-With": "XMLHttpRequest"
      },
      body: JSON.stringify(formData)
    })
    .then(function(res) {
      if (res.ok || res.status === 200) {
        document.getElementById("lq-form-wrap").style.display = "none";
        document.getElementById("lq-success").style.display = "block";
        setTimeout(lqClose, 3500);
      } else {
        throw new Error("Error " + res.status);
      }
    })
    .catch(function(err) {
      var asunto = encodeURIComponent("Solicitud: " + producto);
      var cuerpo = encodeURIComponent(
        "Nombre: " + nombre + " " + apellido + "\n" +
        "Ciudad: " + ciudad + "\n" +
        "Teléfono: " + telefono + "\n" +
        "Email: " + email + "\n" +
        "Fecha: " + fecha + "\n" +
        "Evento: " + evento + "\n" +
        "Producto: " + producto + "\n\n" +
        "Mensaje:\n" + mensaje
      );
      window.location.href = "mailto:TU_CORREO@letsgou.cl?subject=" + asunto + "&body=" + cuerpo;
    })
    .finally(function() {
      btn.disabled = false;
      btn.textContent = "Enviar solicitud";
    });
  };

})();
</script>
